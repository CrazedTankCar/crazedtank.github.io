<script>
    // 1) Expand/collapse insight cards (unique grid expanders)
    function initInsightCards() {
        const previews = document.querySelectorAll('.card-preview');
        previews.forEach(preview => {
            preview.removeEventListener('click', cardHandler);
            preview.addEventListener('click', cardHandler);
        });
    }

    function cardHandler(e) {
        const card = this.closest('.insight-card');
        const detail = card.querySelector('.card-detail');
        const btn = this.querySelector('.expand-btn');
        if (!detail) return;
        const isOpen = detail.classList.contains('open-detail');
        if (isOpen) {
            detail.classList.remove('open-detail');
            if (btn) btn.innerHTML = '+';
        } else {
            detail.classList.add('open-detail');
            if (btn) btn.innerHTML = '−';
        }
        e.stopPropagation();
    }

    // 2) Vertical nav collapsible subtopics
    function initNavSubmenus() {
        const toggles = document.querySelectorAll('[data-toggle]');
        toggles.forEach(toggle => {
            toggle.removeEventListener('click', subToggleHandler);
            toggle.addEventListener('click', subToggleHandler);
        });
    }

    function subToggleHandler(e) {
        const targetId = this.getAttribute('data-toggle');
        const subMenu = document.getElementById(targetId);
        if (subMenu) {
            subMenu.classList.toggle('open-sub');
            const arrow = this.querySelector('.arrow-sub');
            if (arrow) arrow.innerHTML = subMenu.classList.contains('open-sub') ? '▲' : '▼';
        }
        e.stopPropagation();
    }

    // 3) smooth scroll for all nav links (including sub-links)
    document.querySelectorAll('.nav-main-link[href^="#"], .nav-sub-link[href^="#"]').forEach(link => {
        link.addEventListener('click', function(e) {
            const hash = this.getAttribute('href');
            if (hash && hash !== '#') {
                e.preventDefault();
                const targetElem = document.querySelector(hash);
                if (targetElem) {
                    targetElem.scrollIntoView({ behavior: 'smooth', block: 'start' });
                }
            }
        });
    });

    // 4) active link highlight on scroll (optional)
    function setActiveOnScroll() {
        const sections = ['home','clean-water-cat','sanitation-cat','wastewater-cat','conservation-cat','hygiene-cat','credits-footer'];
        const scrollPos = window.scrollY + 150;
        let current = 'home';
        for (let i = sections.length-1; i>=0; i--) {
            const el = document.getElementById(sections[i]);
            if (el && el.offsetTop <= scrollPos) {
                current = sections[i];
                break;
            }
        }
        document.querySelectorAll('.nav-main-link, .nav-sub-link').forEach(link => {
            link.classList.remove('active-link');
            const href = link.getAttribute('href');
            if (href && href.includes(current)) {
                link.classList.add('active-link');
            }
        });
    }

    window.addEventListener('load', () => {
        initInsightCards();
        initNavSubmenus();
        setActiveOnScroll();
        window.addEventListener('scroll', setActiveOnScroll);
    });
</script>
